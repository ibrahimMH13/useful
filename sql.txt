#use when case
SELECT 
    CASE
        WHEN
            (B+A <=A) OR (A+C <=B) OR (A+B <=C)
                THEN "Not A Triangle"
        WHEN
            (A = B) AND (A = C)
                THEN "Equilateral"
        WHEN
            (A=B) OR (B=C) OR (A=C)
                THEN "Isosceles"
        ELSE
              "Scalene"
    END
FROM TRIANGLES;
